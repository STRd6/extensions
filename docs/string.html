    Parse this string as though it is JSON and return the object it represents. If it is not valid JSON returns the string itself.

      parse: () ->
    try
      JSON.parse(this.toString())
    catch e
      this.toString()
  • ¶ 
    #! example
# this is valid json, so an object is returned
'{"a": 3}'.parse()

    Get the file extension of a string.

      extension: ->
    if extension = this.match(/\.([^\.]*)$/, '')?.last()
      extension
    else
      ''
